function doPost(e) { try { var data = JSON.parse(e.postData.contents); if (data.password !== "arrijal123") { return ContentService.createTextOutput(JSON.stringify({"status": "error", "message": "Password salah!"})) .setMimeType(ContentService.MimeType.JSON); } var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var timestamp = new Date(); sheet.appendRow([ timestamp, data.tanggal, data.jenis, data.kategori, data.nominal, data.keterangan, data.link_nota ]); return ContentService.createTextOutput(JSON.stringify({"status": "success"})) .setMimeType(ContentService.MimeType.JSON); } catch(error) { return ContentService.createTextOutput(JSON.stringify({"status": "error", "message": error.toString()})) .setMimeType(ContentService.MimeType.JSON); } } // Tambahan fungsi doGet agar website bisa mengambil data langsung dari Apps Script function doGet(e) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var rows = sheet.getDataRange().getValues(); return ContentService.createTextOutput(JSON.stringify(rows)) .setMimeType(ContentService.MimeType.JSON); }
